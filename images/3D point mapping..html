    clusters = []

    import cv2
import numpy as np
import math

# Function to calculate the distance between two points
def distance(pt1, pt2):
    x1, y1 = pt1
    x2, y2 = pt2
    return math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)

# Function to calculate the angle between two lines
def angle(line1, line2):
    _, _, theta1, _ = line1
    _, _, theta2, _ = line2
    return abs(theta2 - theta1)

# Function to merge nearby lines
def merge_lines(lines, distance_threshold, angle_threshold):
    merged_lines = []
    merged = False

    while not merged:
        merged = True
        lines_copy = lines.copy()
        merged_lines.clear()

        for i, line1 in enumerate(lines_copy):
            for j, line2 in enumerate(lines_copy[i+1:]):
                dist = distance(line1[:2], line2[:2])

                if dist < distance_threshold and angle(line1, line2) < angle_threshold:
                    # Merge lines
                    x1 = min(line1[0], line2[0])
                    y1 = min(line1[1], line2[1])
                    x2 = max(line1[2], line2[2])
                    y2 = max(line1[3], line2[3])

                    merged_lines.append([x1, y1, x2, y2])
                    merged = False

        # Remove merged lines from the original list
        lines = [line for line in lines if line not in merged_lines]

        # Add the merged lines to the original list
        lines.extend(merged_lines)

    return lines
